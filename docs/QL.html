condition(key, value, relation, operator) → {QL}

Source:
  • ql.js, line 750
Since:
  • 2.0.0

Add the influx ql where condition

Examples
const ql = new QL();
ql.measurement = 'http';
ql.condition({
  code: 500,
  spdy: '1',
});
console.info(ql.toSelect());
// => select * from "http" where ("code" = 500 and "spdy" = '1')
const ql = new QL();
ql.measurement = 'http';
ql.condition('spdy', ['1', '2']);
console.info(ql.toSelect());
// => select * from "mydb".."http" where ("spdy" = '1' or "spdy" = '2')
const ql = new QL();
ql.measurement = 'http';
ql.condition({
  code: 500,
  spdy: '1',
}, '!=');
console.info(ql.toSelect());
// => select * from "http" where ("code" != 500 and "spdy" != '1')
